    基于python检测携程网机票价格变化

    安装Python：访问Python官网下载并安装Python。 安装pip：pip是Python的包管理工具，用于安装第三方库。通常Python安装时会自带pip。 安装请求库：打开命令行（Windows中的CMD或PowerShell，macOS或Linux的Terminal），输入以下命令安装requests库，它用于发送HTTP请求： pip install requests

    安装BeautifulSoup库：用于解析HTML页面，输入以下命令安装： bash pip install beautifulsoup4

    airfare query
    import time
import logging
import os
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from selenium.webdriver.edge.service import Service
from webdriver_manager.microsoft import EdgeChromiumDriverManager
from plyer import notification


# 获取当前脚本的目录
script_dir = os.path.dirname(os.path.abspath(__file__))
log_file_path = os.path.join(script_dir, 'airfare_monitor.log')

# 配置日志
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s',
    datefmt='%Y-%m-%d %H:%M:%S',
    filename=log_file_path,  # 使用完整的日志文件路径
    filemode='a',  # 追加模式
    encoding='utf-8'  # 指定日志文件的编码
)

def init_driver():
    service = Service(EdgeChromiumDriverManager().install())
    options = webdriver.EdgeOptions()
    options.use_chromium = True
    options.add_argument("--headless")  # 无头模式
    driver = webdriver.Edge(service=service, options=options)
    return driver

def get_price(driver, url):
    try:
        driver.get(url)
        WebDriverWait(driver, 10).until(
            EC.presence_of_element_located((By.CLASS_NAME, 'head-prices'))
        )
        price_info = driver.find_element(By.CLASS_NAME, 'head-prices')
        price = price_info.text if price_info else '未找到'
        return price
    except Exception as e:
        logging.error(f"Error while getting price: {e}")
        return None

def main():
    driver = init_driver()

    # 定义多个日期组合
    date_combinations = [
        ('2024-09-30', '2024-10-04'),
        ('2024-10-01', '2024-10-05'),
        # 在此添加更多的日期组合
    ]
    # 用于存储每个日期组合的最后价格
    last_prices = {} 

    while True:
        for departure_date, return_date in date_combinations:
            date_str = f"{departure_date},{return_date}"
            url = f'https://www.ly.com/flights/itinerary/roundtrip/ZUH-SIA?date={date_str}&from=%E7%8F%A0%E6%B5%B7&to=%E8%A5%BF%E5%AE%89&refid=2000118609&fromairport=&toairport=&childticket=0,0&flightno=' 
            current_time = time.strftime("%Y-%m-%d %H:%M:%S", time.localtime())
            current_price = get_price(driver, url)

            if current_price != '未找到':
                # 存储当前价格
                last_prices[date_str] = last_prices.get(date_str, current_price)

                # 检查价格是否变化
                if current_price != last_prices[date_str]:
                    last_prices[date_str] = current_price
                    logging.info(f'{current_time}: Price changed to {current_price} for dates {departure_date} to {return_date}')

                    notification.notify(
                        title='票价提醒',
                        message=f'票价有变化，当前票价：{current_price}, 日期：{departure_date} to {return_date}',
                        app_icon=None,
                        timeout=10,
                    )
                else:
                    logging.info(f'{current_time}: Price unchanged {current_price} for dates {departure_date} to {return_date}')
            else:
                logging.warning(f'{current_time}: Price not found for dates {departure_date} to {return_date}')

        time.sleep(300)  # 例如每5分钟检查一次
        with open(log_file_path, 'r', encoding='utf-8') as log_file:
            log_contents = log_file.read()
            #print('Log contents:', log_contents)


if __name__ == '__main__':
    main()

    基于Selenium和Webdriver的网页爬虫

    Python 环境

    安装pip 包管理器

    WebDriver 驱动

    浏览器驱动是运行自动化测试的必要工具，它负责模拟浏览器行为，如点击、输入，还可以动态地显示jscript代码效果。

    • ChromeDriver：Chrome浏览器的驱动，用于控制Chrome浏览器。
    • Microsoft Edge WebDriver：Edge浏览器的驱动，用于控制Edge浏览器。

    1. 安装ChromeDriver pip install webdriver_manager

      pip install --upgrade selenium webdriver-manager

    browser

    浏览器驱动需要与浏览器版本匹配，否则可能会导致无法正确运行自动化测试。（浏览器更最新）

    Selenium Web自动化

    • 用途：用于Web自动化测试和浏览器自动化库。

    • 特点：Selenium 可以模拟用户的行为，如点击、滚动、填写表单等，这些操作可以触发JavaScript和Ajax调用，从而加载和更新页面内容。因此，Selenium 能够处理动态网页，获取页面在用户交互后的状态。

    1. 安装selenium

    pip install selenium

pip install --upgrade selenium

    2.2.4 等待页面加载完成
    try:
    # 等待歌单列表元素加载完成
    wait = WebDriverWait(driver, 300)  # 等待最多30秒

    wait.until(EC.presence_of_all_elements_located((By.CSS_SELECTOR, ".myhkplaylist-bd .album-list .myhklist .mCSB_container ul li")))
    print("歌单列表加载完成！")

    # 获取歌词列表的所有元素
    lyrics_elements = driver.find_elements(By.CSS_SELECTOR, ".myhkplaylist-bd .album-list .myhklist .mCSB_container ul li")
    print("获取到歌词列表！", lyrics_elements)

    # 遍历所有歌词元素，提取文本
    lyrics = [element.get_attribute('textContent').strip() for element in lyrics_elements if element.get_attribute('textContent').strip()]
    print("提取到歌词！", lyrics)

    # 打印歌词
    for line in lyrics:
        print(line)

    # 如果需要，可以将歌词保存到文件
    with open('lyrics.txt', 'w', encoding='utf-8') as file:
        for line in lyrics:
            file.write(line + '\n')

except Exception as e:
    print("发生错误：", e)

finally:
    # 关闭浏览器
    driver.quit()

    注意

    1. 使用try ... except ... finally ... 块来处理元素加载和异常。
    2. 查询前要等待页面加载完成，否则可能获取不到数据。
      1. WebDriverWait: 超时时间
      2. until: 等待要查询的元素加载完成
      3. presence_of_all_elements_located: 指定要加载的全部元素
      4. By.CSS_SELECTOR: CSS选择器，需要通过html定位到要查询的元素及关系，需要具备html基础知识
    3. 提取元素的内容，使用get_attribute('textContent')更稳定
    4. 提取的内容一般是table，使用for循环遍历提取行列数据

    for line in lyrics:

    BeautifulSoup 静态网页解析库

    Python的BeautifulSoup库是一个用于解析HTML和XML文档的库。它常与requests库一起使用来抓取静态的网页内容

    • 用途：用于解析HTML和XML文档。

    • 特点：它提取静态内容，即页面的初始HTML代码。如果内容是通过JavaScript动态加载的，那么在页面加载时 BeautifulSoup 可能无法获取到这些内容，除非它在解析之前已经被渲染到HTML中。

    1. 安装

    安装BeautifulSoup
    pip install beautifulsoup4
pip install lxml  # 或者安装其他解析器如 html5lib

    2. 导入库

    import requests
from bs4 import BeautifulSoup

    3. 抓取网页内容

    url = 'https://www.example.com'
response = requests.get(url)

    4. 解析网页内容

    将获取到的HTML内容传递给BeautifulSoup，同时指定一个解析器。

    soup = BeautifulSoup(html_content, 'lxml')  # 或者使用 'html.parser'

    5. 查找元素

    for link in links:
    print(link.get('href'))  # 获取href属性

print(first_paragraph.text)  # 获取文本内容

# 查找所有的<a>标签
links = soup.find_all('a')

# 查找第一个<p>标签
first_paragraph = soup.find('p')

# 使用CSS选择器
headlines = soup.select('h1, h2, h3')

# 使用属性查找
images = soup.find_all('img', {'src': True})

    6. 提取数据

    # 打印整个文档
print(soup.prettify())

# 保存到文件
with open('output.html', 'w', encoding='utf-8') as file:
    file.write(str(soup))

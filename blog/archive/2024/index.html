    基于python检测携程网机票价格变化

    import time
import logging
import os
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from selenium.webdriver.edge.service import Service
from webdriver_manager.microsoft import EdgeChromiumDriverManager
from plyer import notification


# 获取当前脚本的目录
script_dir = os.path.dirname(os.path.abspath(__file__))
log_file_path = os.path.join(script_dir, 'airfare_monitor.log')

# 配置日志
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s',
    datefmt='%Y-%m-%d %H:%M:%S',
    filename=log_file_path,  # 使用完整的日志文件路径
    filemode='a',  # 追加模式
    encoding='utf-8'  # 指定日志文件的编码
)

def init_driver():
    service = Service(EdgeChromiumDriverManager().install())
    options = webdriver.EdgeOptions()
    options.use_chromium = True
    options.add_argument("--headless")  # 无头模式
    driver = webdriver.Edge(service=service, options=options)
    return driver

def get_price(driver, url):
    try:
        driver.get(url)
        WebDriverWait(driver, 10).until(
            EC.presence_of_element_located((By.CLASS_NAME, 'head-prices'))
        )
        price_info = driver.find_element(By.CLASS_NAME, 'head-prices')
        price = price_info.text if price_info else '未找到'
        return price
    except Exception as e:
        logging.error(f"Error while getting price: {e}")
        return None

def main():
    driver = init_driver()

    # 定义多个日期组合
    date_combinations = [
        ('2024-09-30', '2024-10-04'),
        ('2024-10-01', '2024-10-05'),
        # 在此添加更多的日期组合
    ]
    # 用于存储每个日期组合的最后价格
    last_prices = {} 

    while True:
        for departure_date, return_date in date_combinations:
            date_str = f"{departure_date},{return_date}"
            url = f'https://www.ly.com/flights/itinerary/roundtrip/ZUH-SIA?date={date_str}&from=%E7%8F%A0%E6%B5%B7&to=%E8%A5%BF%E5%AE%89&refid=2000118609&fromairport=&toairport=&childticket=0,0&flightno=' 
            current_time = time.strftime("%Y-%m-%d %H:%M:%S", time.localtime())
            current_price = get_price(driver, url)

            if current_price != '未找到':
                # 存储当前价格
                last_prices[date_str] = last_prices.get(date_str, current_price)

                # 检查价格是否变化
                if current_price != last_prices[date_str]:
                    last_prices[date_str] = current_price
                    logging.info(f'{current_time}: Price changed to {current_price} for dates {departure_date} to {return_date}')

                    notification.notify(
                        title='票价提醒',
                        message=f'票价有变化，当前票价：{current_price}, 日期：{departure_date} to {return_date}',
                        app_icon=None,
                        timeout=10,
                    )
                else:
                    logging.info(f'{current_time}: Price unchanged {current_price} for dates {departure_date} to {return_date}')
            else:
                logging.warning(f'{current_time}: Price not found for dates {departure_date} to {return_date}')

        time.sleep(300)  # 例如每5分钟检查一次
        with open(log_file_path, 'r', encoding='utf-8') as log_file:
            log_contents = log_file.read()
            #print('Log contents:', log_contents)


if __name__ == '__main__':
    main()

    • AnandTech ⭐️⭐️⭐️⭐️⭐️

      【基本信息】AnandTech是一家成立于1997年的美国科技新闻和评测网站，以其深入的技术分析和详尽的评测报告而著称。AnandTech的CPU评测不仅包括性能测试，还包括功耗测试、温度测试、超频潜力分析以及价格与性能比分析。其评测报告通常非常详细，适合对CPU有一定了解的高级用户。

      AnandTech是以组建PC端的配件测评的网站，包括主板、CPU、GPU、SSD等配件测评，也包括笔记本、平板的产品测评，这些测评大多都是基于科技前言的产品。同时，这里也汇聚了全球各大数码发烧友，可以在此相互交流与分享。站内同样搜集了大量发烧友分享的科普文，阅读需要一定的基础知识功底。AnandTech的对比数据基于实际测试的评分得出，而非简单的数据罗列。用户既可以通过型号筛选需要对比的产品，也可以通过测试项来筛选同样的使用场景下，哪款产品的表现更好。

      由于AnandTech不支持中文，并且网站的内容较为“硬核”，对于不太懂硬件的用户来说门槛过高。此外，网站的对比工具入口隐藏较深，需要到页面底部找到“Bench”才能进入。

      【测评范围】笔记本、平板、主板、CPU、GPU、SSD等配件

    Popular name 长 x 高 pixels

    “P”全拼为Progressive译为逐行扫描，几P则表示纵向有多少行像素，比如：720P表示纵向有720行像素、1080P表示纵向有1080行像素。

    “k”表示的是横向排列有多少像素，比如：2k就是视频横向大约有2000列像素、4k就是视频横向大约有4000列像素。

    What is resolution?

    分辨率指图像中的像素数（pixels in an image） 像素越多，图像越详细，分辨率也越高，体积也就越大。

    通常电脑的屏幕分辨率为1920x1080，即1920像素宽，1080像素高。图像像 相比之下，4K的分辨率为3840x2160，即3840像素宽，2160像素高。

    GPU

    简称显卡，它负责渲染图形和处理图像处理任务，用于在显示器中显示图像，包括影像视频、游戏画面、3D模型等。

    显卡厂商

    1. Nvidia
    2. AMD
    3. Intel

    其中截止至2024年9月6日，市场中技术最强的显卡制造厂商是Nvidia，其生产的显卡型号为RTX 4000系列

    NVIDA

    独占技术DLSS

    DLSS 2 本质上是超分辨率技术（Super Resolution），所有RTX显卡都支持。众所周知，显卡在低分辨率条件下帧数很高，在高分辨率下帧数较低。而这项超分辨率技术可以先让显卡渲染低分辨率（比如1080P）图像，通过某种AI算法转化为高分辨率（比如4K）图像输出，以获得更高的帧数。

    DLSS 3本质上是插帧技术（Frame Generation），是RTX 40系显卡独占技术。DLSS 3在两个渲染帧中生成新帧的技术。GPU基于深度学习算法对两个渲染帧的信息来判定新帧，所以这个新帧又被称为AI生成帧，最高能够带来100 %的帧数加成。DLSS 3.5是光线重构技术（Ray Reconstruction），也是所有RTX显卡都支持。通过光线追踪降噪器增强光追的质量，光线追踪降噪器是英伟达的用HPC训练出来的。这项技术只有在光追的时候才能开，在帧数提升上不明显，有的游戏可能没有帧数提升，因为重点是放在光追效果上。以上技术名称都叫DLSS，本质上不是同一项技术，在游戏设置里面甚至是相互独立的选项。唯一共同点就是都和深度学习&人工智能相关，所以都统称为DLSS（Deep Learning Super Sampling）了。DLSS 3.5本质上只是多一个光线重构技术（Ray Reconstruction）。

    显卡性能指标

    如何判断显卡的性能呢？有一些关键的指标可以参考

    1. 帧数（FPS）：在不同的分辨率下显示的画面帧数越高，说明显卡渲染图像的能力又快又强。
    2. 分辨率：显卡能够支持的分辨率越高，显示的画面色彩更丰富更清晰。

    关于帧数，通常认为60帧/秒以上为流畅，120帧/秒以上为高刷。分辨率越高帧数则越低；

    所以一款标准显卡应该具备1080P分辨率，60帧/秒以上。

    一款性能较强的显卡应该支持2080P分辨率，60帧/秒以上。
